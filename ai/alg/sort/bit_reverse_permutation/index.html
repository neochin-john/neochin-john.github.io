def bit_reverse(n, log2N):
    if n >= (2**log2N):
        print("ERR: can't use %d bit to present number %d" % (log2N, n))
        return 0

    n_br = 0
    for _ in range(log2N):
        n_br <<= 1
        if (n & 1 != 0):
            n_br = n_br + 1
        n >>= 1

    return n_br
